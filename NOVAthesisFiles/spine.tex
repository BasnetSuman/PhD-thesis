\newcommand{\@spinetextalin}[1]{%
  \raggedright%
  \IfStrEqCase{\thespine(box,#1,align)}{%
    {l}{\raggedright}%
    {c}{\centering}%
    {r}{\raggedleft}%
  }%
}

\newcommand{\@spinetextcolor}[1]{%
  \ifdatadefinedor{\arg}{spine}({box,#1,textcolor},{box,textcolor}){%
    \color{\thespine(\arg)}%
  }{}%
}
